create table usuarios(
ID_Usuarios INT PK NN,
Nombre VARCHAR (15),
Subnombre VARCHAR (15),
Email VARCHAR (20),
Contraseña VARCHAR (15),
Foto de perfil VARCHAR (45)
Descripcion VARCHAR (50),
Lista de musicas propias VARCHAR (10),
Buscador_ID_Buscador INT PK NN,
ID_Lista de músicas INT PK NN)

create table Musicas(
ID_Músicas INT PK NN,
Título VARCHAR (15),
Genero VARCHAR (10),
ID_Artista VARCHAR (30),
Duracion VARCHAR (45),
Valoraciónes VARCHAR (50),
Artista_ID_Artista INT PK NN,
Valorciónes_ID_usuarios INT PK NN)

create table Artista(
ID_Artista INT PK NN,
Nombre VARCHAR (30),
Imagen VARCHAR (45))

create table Valoraciones(
ID_usuarios INT PK NN,
ID_Valoraciones VARCHAR (45),
ID_Musica VARCHAR (45),
Puntuacion VARCHAR (45),
Comentario VARCHAR (50),
Fecha VARCHAR (8),

create table Buscador(
ID_Buscador INT PK,
Título VARCHAR (15),
Genero VARCHAR (15),
Artista VARCHAR (30),
Valoraciónes VARCHAR (45))

create table Listara de musicas(
ID_Listas de músicas INT PK NN,
ID_Músicas VARCHAR (45),
Nombre VARCHAR (15), 
Descripcion VARCHAR (50))
